\section{Problem}
\begin{enumerate}[label=\thesection.\arabic*.,ref=\thesection.\theenumi]
    \numberwithin{equation}{enumi}
    
    \item Let
    \begin{align}
        x(n) = \cbrak{\underset{\uparrow}{1},2,3,4,2,1} \\
        y(n) + \frac{1}{2}y(n-1) = x(n) + x(n-2)	
    \end{align}
    
    \item Compute X(k), H(k) and y(n) using FFT and IFFT 

\end{enumerate}

\section{Solution}
\begin{enumerate}[label=\thesection.\arabic*.,ref=\thesection.\theenumi]
\numberwithin{equation}{enumi}

\item
\begin{align}
    x(n) = \cbrak{\underset{\uparrow}{1},2,3,4,2,1} \\
    y(n) + \frac{1}{2}y(n-1) = x(n) + x(n-2)	
\end{align}
Impulse Response of the LTI system is the output of the system when Unit Impulse Signal is given as input to the system.\\
\newline
So, Impulse Response of the System is
\begin{align}
    h(n) + \frac{1}{2}h(n-1) = \delta(n) + \delta(n-2)	
\end{align}
$h(n)$ is an IIR Filter.

\item FFT of a Input Signal $x(n)$ is 
\begin{align}
    X(k) = \sum_{n=0}^{N-1} x(n) e^{-j 2 \pi k n / N}, \quad k=0,1, \ldots, N-1
\end{align}

\item FFT of a Impulse Response $h(n)$ is 
\begin{align}
    H(k) = \sum_{n=0}^{N-1} h(n) e^{-j 2 \pi k n / N}, \quad k=0,1, \ldots, N-1
\end{align}

\item then  FFT of output Signal $y(n)$ can be computed  by 
\begin{align}
    Y(k) = X(k)H(k)
\end{align}

\item $y(n)$ can be computed by doing IFFT for $Y(k)$
\begin{align}
    y(n) = \frac{1}{N}\sum_{n=0}^{N-1}Y(k) e^{j 2 \pi n k / N}, \quad k=0,1, \ldots, N-1
\end{align}

\end{enumerate}
